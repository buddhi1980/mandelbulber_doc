% Makro simple image with 75% width
% #1 : file path
% #2 : caption
% #3 : label
\def\simpleImageWithCaptionFullWidth#1#2#3#4{%
	\begin{figure}[{#4}]
		\begin{center}
			\includegraphics[width=1.00\linewidth]{#1}
			\vspace*{-6mm}
			\caption{#2}
			\vspace*{-8mm}
			\label{#3}
		\end{center}
	\end{figure}
}

% Makro two images with two captions

		\begin{center}
			\begin{minipage}[b]{0.4\textwidth}

					\includegraphics[height=0.7\linewidth]{#1}
					\vspace*{-2mm}

					\label{#3}

			\end{minipage}

					\caption{#5}
					\label{#6}
				\end{center}
			\end{minipage}
		\end{center}
	\end{figure}
}
